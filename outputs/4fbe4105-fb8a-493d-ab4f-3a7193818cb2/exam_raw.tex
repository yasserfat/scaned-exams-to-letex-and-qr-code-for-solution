\section*{الجزء الثّاني: (08 نقطات)}

\section*{الوضعيّة:}

قرّرت إحدى البلديّات تهيئة كلٍّ من فناء وقاعة استقبال لروضة أطفال عموميّة قصد حمايتهم من حوادث السّقوط، فخصّصت مبلغا قدره $1500000\, DA$ لإنجاز هذا المشروع.

كلّفت البلديّة أحد المقاولين بإنجاز التّهيئة مع شراء عشب اصطناعيّ لتغطية أرضيّة الفناء وبساط لفرش قاعة الاستقبال.

إذا علمت:
\begin{itemize}
    \item أنّ مساحة أرضيّة الفناء هي $840\, m^2$، وأنّ أرضيّة قاعة الاستقبال على شكل مثلّث قائم طولا ضلعيه القائمين $6\, m$ و $8\, m$.
    \item وأنّ:
    \[
    \begin{cases}
    \text{ثمن } 3\, m^2 \text{ من العشب الاصطناعيّ و} 1\, m^2 \text{ من البساط معًا يقدَّر بـ } 3500\, DA.\\
    \text{ثمن } 1\, m^2 \text{ من العشب الاصطناعيّ و} 2\, m^2 \text{ من البساط معًا يقدَّر بـ } 3000\, DA.
    \end{cases}
    \]
\end{itemize}

\begin{enumerate}
    \item جِد سعر المتر المربّع الواحد من العشب الاصطناعيّ وسعر المتر المربّع الواحد من البساط.
    \item إذا علمت أنّ مصاريف الإنجاز (النّقل وأجرة العمّال) قُدِّرت بـ $20\%$ من المبلغ المُخصَّص لهذا المشروع، ما هو مقدار ربح أو خسارة المقاول؟ مع التّبرير.
\end{enumerate}

\section*{الإجابة النموذجية لموضوع امتحان شهادة التعليم المتوسط دورة: 2023}

\section*{التمرين الأول: (03 نقاط)}

\begin{enumerate}
\item حساب $\text{pgcd}(756 ; 216)$، باستعمال إحدى الطريقتين نجد $\text{pgcd}(756 ; 216) = 108$

إذن: $\dfrac{756}{216} = \dfrac{756 \div 108}{216 \div 108} = \dfrac{7}{2}$ هو كسر غير قابل للاختزال.

\item تبيان أن $B$ يكتب على الشكل $a\sqrt{13}$:
\[
B = \sqrt{117} + 3\sqrt{52} - \sqrt{637} = \sqrt{9 \times 13} + 3\sqrt{4 \times 13} - \sqrt{49 \times 13} = 3\sqrt{13} + 6\sqrt{13} - 7\sqrt{13} = 2\sqrt{13}
\]

\item التحقق من أن: $B \times C = 26\sqrt{3}$
\[
B \times C = 2\sqrt{13} \times \frac{3\sqrt{13}}{\sqrt{3}} = \frac{6 \times 13\sqrt{3} \times \sqrt{3}}{\sqrt{3} \times \sqrt{3}} = \frac{78\sqrt{3}}{3} = 26\sqrt{3}
\]
\end{enumerate}

\section*{التمرين الثاني: (03 نقاط)}

\begin{enumerate}
\item مساحة المربع بدلالة $x$ هي: $(x-2)^2$

مساحة المستطيل بدلالة $x$ هي: $(x-2)(x+2)$

\item تبيان أن: $E + F = 2x(x-2)$
\[
E + F = (x-2)^2 + (x-2)(x+2) = (x-2)(x-2+x+2)
\]
\[
E + F = (x-2)(2x) = 2x(x-2)
\]

\item إيجاد قيمة $x$ التي من أجلها المحيط يساوي $20\,cm$ على الأقل:
\[
5(x-2) + x + 2 + (x+2) - (x-2) \geq 20
\]
\[
5x - 10 + 2x + 4 - x + 2 \geq 20
\]
\[
6x - 4 \geq 20
\]
\[
6x \geq 24
\]
\[
x \geq 4
\]
قيم $x$ هي كل الأطوال الأكبر من أو تساوي $4\,cm$.
\end{enumerate}

\section*{التمرين الثالث: (03 نقاط)}

\begin{enumerate}
\item تبيان أن $(JK) \perp (IM)$ (معناه نبين أن المثلث $JKM$ قائم في $K$)

لدينا في المثلث $JKM$: $JK^2 + KM^2 = 4^2 + 3^2 = 25$ و $JM^2 = 5^2 = 25$

أي $JK^2 + KM^2 = JM^2$، إذن فحسب الخاصية العكسية لفيتاغورس فإن المثلث $JKM$ قائم في $K$

ومنه: $(JK) \perp (IM)$

\item حساب الطول $IK$:

بما أن المثلث $IKJ$ قائم في $K$ و $\hat{I} = 45°$ فإنه متساوي الساقين ومنه: $IK = NJ = 4$

(يمكن الحل باستعمال $\tan 45° = \dfrac{4}{IK}$ منه $IK = 4$)

\item حساب الطول $NK$:

من الشكل لدينا $(NK) // (JM)$، $(JN)$، و$(MK)$ يتقاطعان في $I$

منه حسب خاصية طاليس: $\dfrac{IK}{IM} = \dfrac{NK}{JM}$ أي $\dfrac{4}{7} = \dfrac{NK}{5}$

منه $NK = \dfrac{20}{7}$
\end{enumerate}

\section*{التمرين الرابع: (03 نقاط)}

\begin{enumerate}
\item حساب $x_A$ وحساب $y_B$:

$-2x_A + 3 = 1$ وبالتالي $-2x_A = -2$ و منه: $x_A = 1$

$y_B = -2 \times 2 + 3$ ومنه $y_B = -1$

\item إثبات أن النقط $C$، $O$، $D$ في استقامية:

لدينا: $C(1;2)$، منه $\overrightarrow{OC}\begin{pmatrix}1\\2\end{pmatrix}$

ولدينا: $D(-1;-2)$، منه $\overrightarrow{OD}\begin{pmatrix}-1\\-2\end{pmatrix}$ أي $\overrightarrow{DO}\begin{pmatrix}1\\2\end{pmatrix}$ منه $\overrightarrow{OC} = \overrightarrow{DO}$

ويعني أن $O$ منتصف $[DC]$ إذن النقط $C$، $O$، $D$ في استقامية.

(يمكن توظيف حساب احداثيتي منتصف قطعة مستقيم أو التناظر بالنسبة إلى المبدأ $O$ لإثبات استقامية النقط $D$، $O$، $C$)

\item التمثيل البياني للدالة $f$:

\[
\begin{array}{|c|c|c|}
\hline
x & 0 & 1 \\
\hline
y & 3 & 1 \\
\hline
\end{array}
\]
\end{enumerate}

\section*{الوضعية: (08 نقاط)}

\begin{enumerate}
\item تعيين سعر المتر المربع الواحد من البساط والمتر المربع الواحد من العشب الاصطناعي:

نرمز بـ $X$ لسعر المتر المربع الواحد من البساط

وبـ $Y$ لسعر المتر المربع الواحد من العشب الاصطناعي

فنحصل على الجملة:
\[
\begin{cases} 3x + y = 3500 \quad (1) \\ x + 2y = 3000 \quad (2) \end{cases}
\]

باختيار إحدى الطرائق لحل الجملة نجد أن حل الجملة هو الثنائية $(800 ; 1100)$.

أي: سعر المتر المربع الواحد من البساط هو: $800\,DA$

سعر المتر المربع الواحد من العشب الاصطناعي هو: $1100\,DA$

\item حساب قيمة الربح الذي سيتحصل عليه المقاول:

- حساب مصاريف الإنجاز:
\[
1500000 \times \frac{20}{100} = 300000 \quad \text{أي قيمة المصاريف هي: } 300000\,DA
\]

- حساب مساحة قاعة الاستقبال: $\dfrac{8 \times 6}{2} = \dfrac{48}{2} = 24$ أي تساوي $24\,m^2$

تكلفة شراء البساط والعشب الاصطناعي معًا تساوي:
\[
840 \times 1100 + 24 \times 800 = 943200 \quad \text{أي تساوي: } 943200\,DA
\]

تكلفة المشروع: $943200 + 300000 = 1243200$ أي تساوي: $1243200\,DA$

المقاول ربح ومقدار ربحه: $1500000 - 1243200 = 256800$

أي: $256800\,DA$
\end{enumerate}
