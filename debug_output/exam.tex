\section*{الجزء الأول (12 نقطة)}

\subsection*{التمرين الأول: (02 نقاط)}

إليك العددين $A$ و $B$ حيث:
\[
A = \frac{2}{3} + \frac{7}{3} \times \frac{5}{14} \quad \text{و} \quad B = 2\sqrt{112} - 3\sqrt{28} + 3\sqrt{7}
\]

\begin{enumerate}
    \item اكتب $A$ على شكل كسر غير قابل للاختزال.
    \item اكتب $B$ على الشكل $a\sqrt{7}$ حيث $a$ عدد صحيح.
\end{enumerate}

\subsection*{التمرين الثاني: (03 نقاط)}

$E$ عبارة جبرية حيث:
\[
E = (3x+1)^2 - (x-2)^2
\]

\begin{enumerate}
    \item انشر وبسّط العبارة $E$.
    \item حلّل العبارة $E$ إلى جداء عاملين من الدرجة الأولى.
    \item حلّ المعادلة: $(4x-1)(2x+3) = 0$.
\end{enumerate}

\subsection*{التمرين الثالث: (03 نقاط)}

الشكل المقابل غير مرسوم بالأبعاد الحقيقية.

\begin{center}
\begin{tikzpicture}[scale=0.7]
    % Circle with center O and diameter AB
    \draw (0,0) circle (2.5cm);
    \coordinate (A) at (2.5,0);
    \coordinate (B) at (-2.5,0);
    \coordinate (O) at (0,0);
    % Point M on circle (upper part)
    \coordinate (M) at (-1.3, 2.16);
    % Point E on BM
    \coordinate (E) at (-1.87, 0.86);
    % Point F on AB
    \coordinate (F) at (-0.63, 0);

    \draw (B) -- (A);
    \draw (B) -- (M) -- (A);
    \draw (E) -- (F);

    \fill (A) circle (1.5pt) node[right] {A};
    \fill (B) circle (1.5pt) node[left] {B};
    \fill (O) circle (1.5pt) node[below] {O};
    \fill (M) circle (1.5pt) node[above] {M};
    \fill (E) circle (1.5pt) node[left] {E};
    \fill (F) circle (1.5pt) node[below] {F};

    \node at (-2.7,1.2) {$(C)$};
\end{tikzpicture}
\end{center}

$(C)$ دائرة مركزها النقطة $O$ وقطرها $[AB]$ حيث: $AB = 10$ cm.

$M$ نقطة من $(C)$ حيث: $BM = 6$ cm.

\begin{enumerate}
    \item بيّن نوع المثلث $MBA$ ثم احسب الطول $AM$.
    \item احسب قيس الزاوية $\widehat{MBA}$ ثم أعط مدوّر النتيجة إلى الوحدة بالدرجة.
    \item $E$ نقطة من $[BM]$ حيث $BE = 4{,}2$ cm. المستقيم الذي يشمل $E$ ويعامد $[BM]$ يقطع $[AB]$ في النقطة $F$. احسب الطول $BF$.
\end{enumerate}

\subsection*{التمرين الرابع: (04 نقاط)}

المستوي مزوّد بمعلم متعامد ومتجانس $\left(O;\overrightarrow{OI};\overrightarrow{OJ}\right)$

\begin{enumerate}
    \item علّم النقط $A(1;2)$، $B(5;-2)$، $C(-1;-3)$.
    \item احسب مركّبتي الشعاع $\overrightarrow{BC}$ ثم استنتج الطول $BC$.
    \item احسب إحداثيتي النقطة $M$ منتصف القطعة $[AC]$.
    \item أوجد إحداثيتي النقطة $D$ حيث يكون $\overrightarrow{BM} = \overrightarrow{MD}$ ثم استنتج نوع الرباعي $ABCD$.
\end{enumerate}

\section*{الجزء الثاني: (08 نقاط)}

يريد عمّي محمود إحاطة قطعة أرض مستطيلة الشكل بُعداها $60\,m$ و $42\,m$ بأشجار من نفس النوع بحيث تكون المسافة متساوية وأكبر ما يمكن بين كل شجرتين متتاليتين، على أن يغرس في كل ركن شجرة.

\begin{itemize}
    \item المشتلة التي قصدها عمّي محمود تعرض شجيرات مختلفة، أثمانها من $200\,DA$ إلى $1000\,DA$ حسب نوعيتها. (كلّما كانت الشجيرة أفضل كان ثمنها أكبر).
    \item تكلفة غرس كل شجيرة يمثّل $125\%$ من ثمنها المعروض.
    \item مصاريف النقل $1400\,DA$ مهما كان عدد الشجيرات.
    \item مع عمّي محمود $32000\,DA$.
\end{itemize}

أعطِ القيمة التي لا يمكن أن يتجاوزها ثمن الشجيرة حتّى يتسنّى لعمّي محمود إحاطة هذه القطعة حسب الشروط المذكورة.
